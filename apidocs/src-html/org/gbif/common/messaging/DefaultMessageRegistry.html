039/**
040 * A default implementation of the {@link MessageRegistry} interface where all the messages from this projects {@code
041 * org.gbif.common.messaging.api} package are already preregistered.
042 * <p/>
043 * This class is thread-safe.
044 */
045@ThreadSafe
046public class DefaultMessageRegistry implements MessageRegistry {
047
048  private final Object lock = new Object();
049
050  /* These are the default messages which are the seed for any new instance */
051  private static final ImmutableMap<Class<? extends Message>, String> MESSAGE_TO_EXCHANGE_MAPPING;
052  private static final ImmutableMap<Class<? extends Message>, String> MESSAGE_TO_ROUTING_KEY_MAPPING;
053
054  @GuardedBy("lock")
055  private final ConcurrentMap<Class<? extends Message>, String> exchangeMapping = Maps.newConcurrentMap();
056  @GuardedBy("lock")
057  private final ConcurrentMap<Class<? extends Message>, String> routingKeyMapping = Maps.newConcurrentMap();
058
059  static {
060    MESSAGE_TO_EXCHANGE_MAPPING = ImmutableMap.<Class<? extends Message>, String>builder()
061      .put(CrawlErrorMessage.class, "crawler")
062      .put(CrawlFinishedMessage.class, "crawler")
063      .put(CrawlStartedMessage.class, "crawler")
064      .put(CrawlRequestMessage.class, "crawler")
065      .put(CrawlResponseMessage.class, "crawler")
066      .put(OccurrenceFragmentedMessage.class, "crawler")
067      .put(DwcaDownloadFinishedMessage.class, "crawler")
068      .put(DwcaMetasyncFinishedMessage.class, "crawler")
069      .put(DwcaValidationFinishedMessage.class, "crawler")
070      .put(FragmentPersistedMessage.class, "occurrence")
071      .put(VerbatimPersistedMessage.class, "occurrence")
072      .put(OccurrenceMutatedMessage.class, "occurrence")
073      .put(DeleteOccurrenceMessage.class, "occurrence")
074      .put(DeleteDataResourceOccurrencesMessage.class, "occurrence")
075      .put(DeleteDatasetOccurrencesMessage.class, "occurrence")
076      .put(ParseFragmentMessage.class, "occurrence")
077      .put(ParseDatasetMessage.class, "occurrence")
078      .put(InterpretVerbatimMessage.class, "occurrence")
079      .put(InterpretDatasetMessage.class, "occurrence")
080      .put(RegistryChangeMessage.class, "registry")
081      .put(StartCrawlMessage.class, "registry")
082      .put(StartMetasyncMessage.class, "registry")
083      .build();
084
085    MESSAGE_TO_ROUTING_KEY_MAPPING = ImmutableMap.<Class<? extends Message>, String>builder()
086      .put(CrawlErrorMessage.class, "crawl.error")
087      .put(CrawlFinishedMessage.class, "crawl.finished")
088      .put(CrawlStartedMessage.class, "crawl.started")
089      .put(CrawlRequestMessage.class, "crawl.request")
090      .put(CrawlResponseMessage.class, "crawl.response")
091      .put(DwcaDownloadFinishedMessage.class, DwcaDownloadFinishedMessage.ROUTING_KEY)
092      .put(DwcaMetasyncFinishedMessage.class, DwcaMetasyncFinishedMessage.ROUTING_KEY)
093      .put(DwcaValidationFinishedMessage.class, DwcaValidationFinishedMessage.ROUTING_KEY)
094      .put(OccurrenceFragmentedMessage.class, "crawler.fragment.new")
095      .put(FragmentPersistedMessage.class, "occurrence.fragment.persisted")
096      .put(VerbatimPersistedMessage.class, "occurrence.verbatim.persisted")
097      .put(OccurrenceMutatedMessage.class, "occurrence.interpreted.mutated")
098      .put(DeleteOccurrenceMessage.class, "occurrence.delete.occurrence")
099      .put(DeleteDataResourceOccurrencesMessage.class, "occurrence.delete.dataresource")
100      .put(DeleteDatasetOccurrencesMessage.class, "occurrence.delete.dataset")
101      .put(ParseFragmentMessage.class, "occurrence.parse.occurrence")
102      .put(ParseDatasetMessage.class, "occurrence.parse.dataset")
103      .put(InterpretVerbatimMessage.class, "occurrence.interpret.occurrence")
104      .put(InterpretDatasetMessage.class, "occurrence.interpret.dataset")
105      .put(RegistryChangeMessage.class, "registry.change.#")
106      .put(StartCrawlMessage.class, "crawl.start")
107      .put(StartMetasyncMessage.class, StartMetasyncMessage.ROUTING_KEY)
108      .build();
109  }
110
111  public DefaultMessageRegistry() {
112    exchangeMapping.putAll(MESSAGE_TO_EXCHANGE_MAPPING);
113    routingKeyMapping.putAll(MESSAGE_TO_ROUTING_KEY_MAPPING);
114  }
115
116  @Override
117  public Optional<String> getExchange(Class<? extends Message> message) {
118    checkNotNull(message, "message can't be null");
119
120    return Optional.fromNullable(exchangeMapping.get(message));
121  }
122
123  @Override
124  public Optional<String> getGenericRoutingKey(Class<? extends Message> message) {
125    checkNotNull(message, "message can't be null");
126
127    return Optional.fromNullable(routingKeyMapping.get(message));
128  }
129
130  @Override
131  public ImmutableSet<Class<? extends Message>> getRegisteredMessages() {
132    synchronized (lock) {
133      return ImmutableSet.copyOf(exchangeMapping.keySet());
134    }
135  }
136
137  /**
138   * Used to register a new message or change information about an existing one. No special care is taken to protect
139   * the default set of messages.
140   *
141   * @param message    to register
142   * @param exchange   it uses
143   * @param routingKey it uses
144   */
145  @Override
146  public void register(Class<? extends Message> message, String exchange, String routingKey) {
147    checkNotNull(message, "message can't be null");
148    checkNotNull(exchange, "exchange can't be null");
149    checkNotNull(routingKey, "routingKey can't be null");
150
151    synchronized (lock) {
152      exchangeMapping.put(message, exchange);
153      routingKeyMapping.put(message, routingKey);
154    }
155  }
156
157  /**
158   * Deletes information about a message from the registry. No special care is taken to protect the default set of
159   * messages.
160   *
161   * @param message to unregister
162   */
163  @Override
164  public void unregister(Class<? extends Message> message) {
165    checkNotNull(message, "message can't be null");
166
167    synchronized (lock) {
168      exchangeMapping.remove(message);
169      routingKeyMapping.remove(message);
170    }
171  }
172
173  /**
174   * Deletes information about all message from the registry. No special care is taken to protect the default set of
175   * messages.
176   */
177  @Override
178  public void clear() {
179    synchronized (lock) {
180      exchangeMapping.clear();
181      routingKeyMapping.clear();
182    }
183  }
184}
