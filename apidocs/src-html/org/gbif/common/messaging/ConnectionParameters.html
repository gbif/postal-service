001package org.gbif.common.messaging;
002
003import com.google.inject.Inject;
004import com.google.inject.name.Named;
005import com.rabbitmq.client.ConnectionFactory;
006
007import static com.google.common.base.Preconditions.checkArgument;
008import static com.google.common.base.Preconditions.checkNotNull;
009
010/**
011 * Encapsulates all connection parameters necessary to connect to RabbitMQ.
012 */
013public class ConnectionParameters {
014
015  private final String host;
016  private final int port;
017  private final String username;
018  private final String password;
019  private final String virtualHost;
020
021  @Inject
022  public ConnectionParameters(
023    @Named("hostname") String host,
024    @Named("port") int port,
025    @Named("username") String username,
026    @Named("password") String password,
027    @Named("virtualhost") String virtualHost
028  ) {
029    this.host = checkNotNull(host, "host can't be null");
030    this.port = port;
031    this.username = checkNotNull(username, "username can't be null");
032    this.password = checkNotNull(password, "password can't be null");
033    this.virtualHost = checkNotNull(virtualHost, "virtualHost can't be null");
034
035    checkArgument(!host.isEmpty(), "host can't be empty");
036    checkArgument(port > 0, "port has to be greater than zero");
037    checkArgument(!username.isEmpty(), "username can't be empty");
038    checkArgument(!password.isEmpty(), "password can't be empty");
039    checkArgument(!virtualHost.isEmpty(), "virtualHost can't be empty");
040  }
041
042  public String getHost() {
043    return host;
044  }
045
046  public int getPort() {
047    return port;
048  }
049
050  public String getUsername() {
051    return username;
052  }
053
054  public String getPassword() {
055    return password;
056  }
057
058  public String getVirtualHost() {
059    return virtualHost;
060  }
061
062  ConnectionFactory getConnectionFactory() {
063    ConnectionFactory connectionFactory = new ConnectionFactory();
064    connectionFactory.setHost(host);
065    connectionFactory.setPort(port);
066    connectionFactory.setUsername(username);
067    connectionFactory.setPassword(password);
068    connectionFactory.setVirtualHost(virtualHost);
069    return connectionFactory;
070  }
071
072}
