001package org.gbif.common.messaging;
002
003import org.gbif.common.messaging.api.Message;
004import org.gbif.common.messaging.api.MessagePublisher;
005import org.gbif.common.messaging.api.MessageRegistry;
006
007import java.io.IOException;
008import java.util.concurrent.ConcurrentLinkedQueue;
009import javax.annotation.concurrent.ThreadSafe;
010
011import com.fasterxml.jackson.datatype.guava.GuavaModule;
012import com.google.common.base.Optional;
013import com.google.common.collect.Queues;
014import com.rabbitmq.client.Channel;
015import com.rabbitmq.client.Connection;
016import com.rabbitmq.client.MessageProperties;
017import org.codehaus.jackson.map.ObjectMapper;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021import static com.google.common.base.Preconditions.checkArgument;
022import static com.google.common.base.Preconditions.checkNotNull;
023
024/**
025 * This class can be used to publish messages easily.
026 */
027@ThreadSafe
028public class DefaultMessagePublisher implements MessagePublisher {
029
030  private static final Logger LOG = LoggerFactory.getLogger(DefaultMessagePublisher.class);
031  private static final int NUMBER_OF_RETRIES = 3;
032  private final Connection connection;
033  private final MessageRegistry registry;
034  private final ObjectMapper mapper;
035  /**
036   * The pool of channels available for use. Clients will reuse available channels, but when the pool is exhausted,
037   * new channels are created. This is unbounded in size and dependent on consumer load (e.g. concurrent threads).
038   */
039  private final ConcurrentLinkedQueue<Channel> channelPool = Queues.newConcurrentLinkedQueue();
040
041  /**
042   * Convenience constructor that uses a default {@link ObjectMapper} and the {@link DefaultMessageRegistry}.
043   *
044   * @param connectionParameters to use
045   *
046   * @see #DefaultMessagePublisher(ConnectionParameters, MessageRegistry, ObjectMapper)
047   */
048  public DefaultMessagePublisher(ConnectionParameters connectionParameters) throws IOException {
049    this(connectionParameters, new DefaultMessageRegistry(), new ObjectMapper());
050  }
051
052  /**
053   * Builds a new MessagingService with the provided components, and declares all the necessary exchanges as topic
054   * based.
055   *
056   * @param connectionParameters to use
057   * @param registry             to use
058   * @param mapper               to use
059   */
060  public DefaultMessagePublisher(
061    ConnectionParameters connectionParameters, MessageRegistry registry, ObjectMapper mapper
062  ) throws IOException {
063    checkNotNull(connectionParameters, "connectionParameters can't be null");
064    this.mapper = checkNotNull(mapper, "mapper can't be null");
065    this.registry = checkNotNull(registry, "registry can't be null");
066
067    mapper.registerModule(new GuavaModule());
068    connection = connectionParameters.getConnectionFactory().newConnection();
069    declareAllExchanges(registry, connection);
070  }
071
072  @Override
073  public void send(Message message) throws IOException {
074    checkNotNull(message, "message can't be null");
075
076    Optional<String> exchange = registry.getExchange(message.getClass());
077    checkArgument(exchange.isPresent(), "No exchange found for Message");
078
079    send(message, exchange.get());
080  }
081
082  @Override
083  public void send(Message message, String exchange) throws IOException {
084    checkNotNull(message, "message can't be null");
085    checkNotNull(exchange, "exchange can't be null");
086
087    String routingKey = message.getRoutingKey();
088    send(message, exchange, routingKey);
089  }
090
091  @Override
092  public void send(Object message, String exchange, String routingKey) throws IOException {
093    checkNotNull(message, "message can't be null");
094    checkNotNull(exchange, "exchange can't be null");
095    checkNotNull(routingKey, "routingKey can't be null");
096
097    byte[] data = mapper.writeValueAsBytes(message);
098    LOG.debug("Sending message of type [{}] to exchange [{}] using routing key [{}]",
099              message.getClass().getSimpleName(),
100              exchange,
101              routingKey);
102
103    for (int attempt = 1; attempt <= NUMBER_OF_RETRIES; attempt++) {
104      Channel channel = provideChannel();
105
106      try {
107        channel.basicPublish(exchange, routingKey, MessageProperties.TEXT_PLAIN, data);
108        // We're not releasing this in a finally block because we assume the channel is "bad" if an exception occurred
109        releaseChannel(channel);
110        return;
111      } catch (IOException e) {
112        if (attempt >= NUMBER_OF_RETRIES) {
113          LOG.warn("Tried sending message but failed {} times, aborting", attempt);
114          throw e;
115        }
116        LOG.debug("Failed sending message, retrying");
117      }
118    }
119
120  }
121
122  /**
123   * Declares all exchanges used by the registry as <em>topic</em> exchanges.
124   *
125   * @param registry   to get the registered messages and exchanges from
126   * @param connection used to declare the exchanges
127   *
128   * @throws IOException if there was a problem declaring an exchange. This might leave the exchanges partially
129   *                     declared
130   */
131  private void declareAllExchanges(MessageRegistry registry, Connection connection) throws IOException {
132    Channel channel = null;
133    try {
134      channel = connection.createChannel();
135      for (Class<? extends Message> message : registry.getRegisteredMessages()) {
136        channel.exchangeDeclare(registry.getExchange(message).get(), "topic", true);
137      }
138    } finally {
139      if (channel != null) {
140        channel.close();
141      }
142    }
143  }
144
145  /**
146   * Callers must return the channel using {@link #releaseChannel(Channel channel) releaseChannel}.
147   *
148   * @return a channel to reuse, or a new one if none are available
149   */
150  private Channel provideChannel() throws IOException {
151    Channel channel = channelPool.poll();
152    if (channel == null || !channel.isOpen()) { // none for reuse
153      LOG.debug("No pooled channels available, creating a new one");
154      channel = connection.createChannel();
155    }
156    return channel;
157  }
158
159  /**
160   * Returns the channels to the pool for reuse.
161   *
162   * @param channel To free up for another use
163   */
164  private void releaseChannel(Channel channel) {
165    // only reuse open channels
166    if (channel.isOpen()) {
167      channelPool.add(channel);
168      LOG.debug("Channel returned to the pool. Available channels for reuse: {}", channelPool.size());
169    } else {
170      LOG.debug("Discarding channel since it is closed. Available channels for reuse: {}", channelPool.size());
171    }
172  }
173
174  /**
175   * Does nothing.
176   */
177  @Override
178  public void close() {
179  }
180}
