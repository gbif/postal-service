001package org.gbif.common.messaging.api;
002
003import com.google.common.base.Optional;
004import com.google.common.collect.ImmutableSet;
005
006/**
007 * Used to register default exchanges and routing keys for Messages.
008 */
009public interface MessageRegistry {
010
011  /**
012   * Gets the default exchange used for this type of message.
013   *
014   * @param message to get the exchange for
015   *
016   * @return an Optional containing a exchange name if it is available
017   */
018  Optional<String> getExchange(Class<? extends Message> message);
019
020  /**
021   * Gets a routing key that can be used to receive all messages of this type no matter if they are using a more
022   * specialized routing key or not.
023   * <p/>
024   * Example: We could use one message class to announce new or updated occurrences. They could use the routing keys
025   * {@code occurrence.new} and {@code occurrence.updated} respectively. This method could return {@code occurrence.*}.
026   *
027   * @param message to get the generic routing key for
028   *
029   * @return an Optional containing a routing key that can be used to bind a queue to a topic exchange
030   */
031  Optional<String> getGenericRoutingKey(Class<? extends Message> message);
032
033  /**
034   * Returns an immutable set of all registered message classes.
035   *
036   * @return immutable set of registered message classes
037   */
038  ImmutableSet<Class<? extends Message>> getRegisteredMessages();
039
040  /**
041   * Registers a new message in this registry. If any information did already exist about this message we overwrite it.
042   *
043   * @param message    to register
044   * @param exchange   it uses
045   * @param routingKey it uses
046   */
047  void register(Class<? extends Message> message, String exchange, String routingKey);
048
049  /**
050   * Deletes any information this registry might have about a message type. We ignore this request if we don't have any
051   * information about the message.
052   *
053   * @param message to unregister
054   */
055  void unregister(Class<? extends Message> message);
056
057  /**
058   * Unregisters all currently registered messages.
059   */
060  void clear();
061
062}
