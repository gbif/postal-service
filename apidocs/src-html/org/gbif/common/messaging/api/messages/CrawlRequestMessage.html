001package org.gbif.common.messaging.api.messages;
002
003import java.util.Arrays;
004import java.util.UUID;
005
006import com.google.common.base.Objects;
007import org.codehaus.jackson.annotate.JsonCreator;
008import org.codehaus.jackson.annotate.JsonProperty;
009
010import static com.google.common.base.Preconditions.checkArgument;
011import static com.google.common.base.Preconditions.checkNotNull;
012
013/**
014 * We send this message every time before we make a request to an endpoint.
015 */
016public class CrawlRequestMessage implements DatasetBasedMessage {
017
018  private final UUID datasetUuid;
019
020  private final int attempt;
021
022  private final int requestTry;
023
024  private final byte[] request;
025
026  private final String status;
027
028  @JsonCreator
029  public CrawlRequestMessage(
030    @JsonProperty("datasetUuid") UUID datasetUuid,
031    @JsonProperty("attempt") int attempt,
032    @JsonProperty("requestTry") int requestTry,
033    @JsonProperty("request") byte[] request,
034    @JsonProperty("status") String status
035
036  ) {
037    this.datasetUuid = checkNotNull(datasetUuid, "datasetUuid can't be null");
038    checkArgument(attempt > 0, "attempt has to be greater than 0");
039    this.attempt = attempt;
040    checkArgument(requestTry > 0, "requestTry has to be greater than 0");
041    this.requestTry = requestTry;
042    this.request = checkNotNull(request, "request can't be null");
043    this.status = checkNotNull(status, "status can't be null");
044  }
045
046  @Override
047  public UUID getDatasetUuid() {
048    return datasetUuid;
049  }
050
051  public int getAttempt() {
052    return attempt;
053  }
054
055  public int getRequestTry() {
056    return requestTry;
057  }
058
059  public byte[] getRequest() {
060    return request;
061  }
062
063  /**
064   * Returns a human-readable status that includes information about what is being crawled. In Crawler terms this would
065   * be the current CrawlContext.
066   *
067   * @return human-readable status
068   */
069  public String getStatus() {
070    return status;
071  }
072
073  @Override
074  public String getRoutingKey() {
075    return "crawl.request";
076  }
077
078  @Override
079  public boolean equals(Object obj) {
080    if (this == obj) {
081      return true;
082    }
083    if (!(obj instanceof CrawlRequestMessage)) {
084      return false;
085    }
086
087    final CrawlRequestMessage other = (CrawlRequestMessage) obj;
088    return Objects.equal(this.datasetUuid, other.datasetUuid)
089           && Objects.equal(this.attempt, other.attempt)
090           && Objects.equal(this.requestTry, other.requestTry)
091           && Arrays.equals(this.request, other.request)
092           && Objects.equal(this.status, other.status);
093  }
094
095  @Override
096  public int hashCode() {
097    return Objects.hashCode(datasetUuid, attempt, requestTry, Arrays.hashCode(request), status);
098  }
099
100  @Override
101  public String toString() {
102    return Objects.toStringHelper(this)
103      .add("datasetUuid", datasetUuid)
104      .add("attempt", attempt)
105      .add("requestTry", requestTry)
106      .add("request", request)
107      .add("status", status)
108      .toString();
109  }
110
111}
