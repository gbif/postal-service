001package org.gbif.common.messaging.api.messages;
002
003import org.gbif.common.messaging.api.Message;
004
005import java.util.UUID;
006
007import com.google.common.base.Objects;
008import org.codehaus.jackson.annotate.JsonCreator;
009import org.codehaus.jackson.annotate.JsonProperty;
010
011import static com.google.common.base.Preconditions.checkNotNull;
012
013/**
014 * Message to send to request a new metadata synchronisation of an Installation.
015 */
016public class StartMetasyncMessage implements Message {
017
018  public static final String ROUTING_KEY = "metasync.start";
019  private final UUID installationKey;
020
021  @JsonCreator
022  public StartMetasyncMessage(@JsonProperty("installationKey") UUID installationKey) {
023    this.installationKey = checkNotNull(installationKey, "installationKey can't be null");
024  }
025
026  public UUID getInstallationKey() {
027    return installationKey;
028  }
029
030  @Override
031  public String getRoutingKey() {
032    return ROUTING_KEY;
033  }
034
035  @Override
036  public int hashCode() {
037    return Objects.hashCode(installationKey);
038  }
039
040  @Override
041  public boolean equals(Object obj) {
042    if (this == obj) {
043      return true;
044    }
045    if (!(obj instanceof StartMetasyncMessage)) {
046      return false;
047    }
048
049    StartMetasyncMessage other = (StartMetasyncMessage) obj;
050    return Objects.equal(this.installationKey, other.installationKey);
051  }
052
053  @Override
054  public String toString() {
055    return Objects.toStringHelper(this).add("installationKey", installationKey).toString();
056  }
057
058}
