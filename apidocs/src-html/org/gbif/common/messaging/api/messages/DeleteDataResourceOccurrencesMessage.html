001package org.gbif.common.messaging.api.messages;
002
003import org.gbif.common.messaging.api.Message;
004
005import com.google.common.base.Objects;
006import org.codehaus.jackson.annotate.JsonCreator;
007import org.codehaus.jackson.annotate.JsonProperty;
008
009/**
010 * This message instructs the occurrence deletion service to delete all occurrence records for the given data resource
011 * id (the legacy id from the MySQL portal).
012 */
013public class DeleteDataResourceOccurrencesMessage implements Message {
014
015  private final int dataResourceId;
016
017  @JsonCreator
018  public DeleteDataResourceOccurrencesMessage(@JsonProperty("dataResourceId") int dataResourceId) {
019    this.dataResourceId = dataResourceId;
020  }
021
022  @Override
023  public String getRoutingKey() {
024    return "occurrence.delete.dataresource";
025  }
026
027  public int getDataResourceId() {
028    return dataResourceId;
029  }
030
031  @Override
032  public int hashCode() {
033    return Objects.hashCode(dataResourceId);
034  }
035
036  @Override
037  public boolean equals(Object obj) {
038    if (this == obj) {
039      return true;
040    }
041    if (obj == null || getClass() != obj.getClass()) {
042      return false;
043    }
044    final DeleteDataResourceOccurrencesMessage other = (DeleteDataResourceOccurrencesMessage) obj;
045    return Objects.equal(this.dataResourceId, other.dataResourceId);
046  }
047}
