001package org.gbif.common.messaging.api.messages;
002
003import org.gbif.api.model.crawler.FinishReason;
004
005import java.util.UUID;
006
007import com.google.common.base.Objects;
008import org.codehaus.jackson.annotate.JsonCreator;
009import org.codehaus.jackson.annotate.JsonProperty;
010
011import static com.google.common.base.Preconditions.checkArgument;
012import static com.google.common.base.Preconditions.checkNotNull;
013
014/**
015 * We send this every time we finish a crawl.
016 */
017public class CrawlFinishedMessage implements DatasetBasedMessage {
018
019  private final UUID datasetUuid;
020  private final int attempt;
021  private final int totalRecordCount;
022  private final FinishReason reason;
023
024  @JsonCreator
025  public CrawlFinishedMessage(
026    @JsonProperty("datasetUuid") UUID datasetUuid,
027    @JsonProperty("attempt") int attempt,
028    @JsonProperty("totalRecordCount") int totalRecordCount,
029    @JsonProperty("reason") FinishReason reason
030  ) {
031    this.datasetUuid = checkNotNull(datasetUuid, "datasetUuid can't be null");
032    checkArgument(attempt > 0, "attempt has to be greater than 0");
033    this.attempt = attempt;
034    checkArgument(totalRecordCount >= 0, "totalRecordCount has to be greater than or equal to 0");
035    this.totalRecordCount = totalRecordCount;
036    this.reason = checkNotNull(reason, "reason can't be null");
037  }
038
039  public int getAttempt() {
040    return attempt;
041  }
042
043  @Override
044  public UUID getDatasetUuid() {
045    return datasetUuid;
046  }
047
048  public FinishReason getReason() {
049    return reason;
050  }
051
052  public int getTotalRecordCount() {
053    return totalRecordCount;
054  }
055
056  @Override
057  public String getRoutingKey() {
058    return "crawl.finished";
059  }
060
061  @Override
062  public boolean equals(Object obj) {
063    if (this == obj) {
064      return true;
065    }
066    if (!(obj instanceof CrawlFinishedMessage)) {
067      return false;
068    }
069
070    final CrawlFinishedMessage other = (CrawlFinishedMessage) obj;
071    return Objects.equal(this.datasetUuid, other.datasetUuid)
072           && Objects.equal(this.attempt, other.attempt)
073           && Objects.equal(this.totalRecordCount, other.totalRecordCount)
074           && Objects.equal(this.reason, other.reason);
075  }
076
077  @Override
078  public int hashCode() {
079    return Objects.hashCode(datasetUuid, attempt, totalRecordCount, reason);
080  }
081
082  @Override
083  public String toString() {
084    return Objects.toStringHelper(this)
085      .add("datasetUuid", datasetUuid)
086      .add("attempt", attempt)
087      .add("totalRecordCount", totalRecordCount)
088      .add("reason", reason)
089      .toString();
090  }
091
092}
