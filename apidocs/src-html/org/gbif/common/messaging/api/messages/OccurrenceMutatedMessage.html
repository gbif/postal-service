001package org.gbif.common.messaging.api.messages;
002
003import org.gbif.api.model.occurrence.Occurrence;
004import org.gbif.api.vocabulary.OccurrencePersistenceStatus;
005
006import java.util.UUID;
007import javax.annotation.Nullable;
008
009import com.google.common.base.Objects;
010import org.codehaus.jackson.annotate.JsonCreator;
011import org.codehaus.jackson.annotate.JsonProperty;
012
013import static com.google.common.base.Preconditions.checkArgument;
014import static com.google.common.base.Preconditions.checkNotNull;
015
016/**
017 * The message sent whenever an "interpreted" occurrence has changed (either NEW, UPDATED, or DELETED). An interpreted
018 * occurrence is the result of interpreting the values of a verbatim occurrence.
019 */
020public class OccurrenceMutatedMessage implements DatasetBasedMessage {
021
022  private final UUID datasetUuid;
023  // the crawl attempt id - only meaningful for NEW and UPDATE
024  private final Integer crawlAttempt;
025  private final OccurrencePersistenceStatus status;
026  private final Occurrence oldOccurrence;
027  private final Occurrence newOccurrence;
028
029  // only used for deletions
030  // if this is a deletion, the reason it has been deleted
031  private final OccurrenceDeletionReason deletionReason;
032  // the last crawl attempt in which this occurrence was seen
033  private final Integer crawlAttemptLastSeen;
034  // the latest crawl attempt for this occurrence's dataset
035  private final Integer latestCrawlAttemptForDataset;
036
037  @JsonCreator
038  public OccurrenceMutatedMessage(@JsonProperty("datasetUuid") UUID datasetUuid,
039    @JsonProperty("status") OccurrencePersistenceStatus status,
040    @Nullable @JsonProperty("crawlAttempt") Integer crawlAttempt,
041    @Nullable @JsonProperty("oldOccurrence") Occurrence oldOccurrence,
042    @Nullable @JsonProperty("newOccurrence") Occurrence newOccurrence,
043    @Nullable @JsonProperty("deletionReason") OccurrenceDeletionReason deletionReason,
044    @Nullable @JsonProperty("crawlAttemptLastSeen") Integer crawlAttemptLastSeen,
045    @Nullable @JsonProperty("latestCrawlAttemptForDataset") Integer latestCrawlAttemptForDataset) {
046    this.datasetUuid = checkNotNull(datasetUuid, "datasetUuid can't be null");
047    this.status = checkNotNull(status, "status can't be null");
048    switch (status) {
049      case NEW:
050        checkNotNull(newOccurrence, "newOccurrence can't be null");
051        checkArgument(crawlAttempt != null && crawlAttempt > 0, "attempt must be greater than 0");
052        break;
053      case UPDATED:
054        checkNotNull(newOccurrence, "newOccurrence can't be null for updates");
055        checkNotNull(oldOccurrence, "oldOccurrence can't be null for updates");
056        checkArgument(crawlAttempt != null && crawlAttempt > 0, "attempt must be greater than 0");
057        break;
058      case DELETED:
059        checkNotNull(oldOccurrence, "oldOccurrence can't be null for deletes");
060        checkNotNull(deletionReason, "deletionReason can't be null for deletes");
061        if (deletionReason == OccurrenceDeletionReason.NOT_SEEN_IN_LAST_CRAWL) {
062          checkArgument(crawlAttemptLastSeen != null && crawlAttemptLastSeen > 0,
063            "crawlAttemptLastSeen must be greater than 0");
064          checkArgument(latestCrawlAttemptForDataset != null && latestCrawlAttemptForDataset > 0,
065            "latestCrawlAttemptForDataset must be greater than 0");
066        }
067        break;
068      case UNCHANGED: // should never be seen
069        break;
070    }
071    this.crawlAttempt = crawlAttempt;
072    this.newOccurrence = newOccurrence;
073    this.oldOccurrence = oldOccurrence;
074    this.deletionReason = deletionReason;
075    this.crawlAttemptLastSeen = crawlAttemptLastSeen;
076    this.latestCrawlAttemptForDataset = latestCrawlAttemptForDataset;
077  }
078
079  public static OccurrenceMutatedMessage buildNewMessage(UUID datasetUuid, Occurrence newOccurrence, int crawlAttempt) {
080    return new OccurrenceMutatedMessage(datasetUuid, OccurrencePersistenceStatus.NEW, crawlAttempt, null, newOccurrence,
081      null, null, null);
082  }
083
084  public static OccurrenceMutatedMessage buildUpdateMessage(UUID datasetUuid, Occurrence oldOccurrence,
085    Occurrence newOccurrence, int crawlAttempt) {
086    return new OccurrenceMutatedMessage(datasetUuid, OccurrencePersistenceStatus.UPDATED, crawlAttempt, oldOccurrence,
087      newOccurrence, null, null, null);
088  }
089
090  public static OccurrenceMutatedMessage buildDeleteMessage(UUID datasetUuid, Occurrence oldOccurrence,
091    OccurrenceDeletionReason reason, Integer crawlAttemptLastSeen, Integer lastCrawlAttemptForDataset) {
092    return new OccurrenceMutatedMessage(datasetUuid, OccurrencePersistenceStatus.DELETED, null, oldOccurrence, null,
093      reason, crawlAttemptLastSeen, lastCrawlAttemptForDataset);
094  }
095
096  @Override
097  public String getRoutingKey() {
098    return "occurrence.interpreted.mutated";
099  }
100
101  @Override
102  public UUID getDatasetUuid() {
103    return datasetUuid;
104  }
105
106  public Integer getCrawlAttempt() {
107    return crawlAttempt;
108  }
109
110  public OccurrencePersistenceStatus getStatus() {
111    return status;
112  }
113
114  public Occurrence getOldOccurrence() {
115    return oldOccurrence;
116  }
117
118  public Occurrence getNewOccurrence() {
119    return newOccurrence;
120  }
121
122  public OccurrenceDeletionReason getDeletionReason() {
123    return deletionReason;
124  }
125
126  public Integer getCrawlAttemptLastSeen() {
127    return crawlAttemptLastSeen;
128  }
129
130  public Integer getLatestCrawlAttemptForDataset() {
131    return latestCrawlAttemptForDataset;
132  }
133
134  @Override
135  public int hashCode() {
136    return Objects
137      .hashCode(datasetUuid, crawlAttempt, status, oldOccurrence, newOccurrence, deletionReason, crawlAttemptLastSeen,
138        latestCrawlAttemptForDataset);
139  }
140
141  @Override
142  public boolean equals(Object obj) {
143    if (this == obj) {
144      return true;
145    }
146    if (obj == null || getClass() != obj.getClass()) {
147      return false;
148    }
149    final OccurrenceMutatedMessage other = (OccurrenceMutatedMessage) obj;
150    return Objects.equal(this.datasetUuid, other.datasetUuid) && Objects.equal(this.crawlAttempt, other.crawlAttempt)
151           && Objects.equal(this.status, other.status) && Objects.equal(this.oldOccurrence, other.oldOccurrence)
152           && Objects.equal(this.newOccurrence, other.newOccurrence) && Objects
153      .equal(this.deletionReason, other.deletionReason) && Objects
154             .equal(this.crawlAttemptLastSeen, other.crawlAttemptLastSeen) && Objects
155             .equal(this.latestCrawlAttemptForDataset, other.latestCrawlAttemptForDataset);
156  }
157
158  @Override
159  public String toString() {
160    return Objects.toStringHelper(this).add("datasetUuid", datasetUuid).add("crawlAttempt", crawlAttempt)
161      .add("status", status).add("oldOccurrence", oldOccurrence).add("newOccurrence", newOccurrence)
162      .add("deletionReason", deletionReason).add("crawlAttemptLastSeen", crawlAttemptLastSeen)
163      .add("latestCrawlAttemptForDataset", latestCrawlAttemptForDataset).toString();
164  }
165}
