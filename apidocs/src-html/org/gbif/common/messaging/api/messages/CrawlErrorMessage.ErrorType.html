001package org.gbif.common.messaging.api.messages;
002
003import java.util.UUID;
004import javax.annotation.Nullable;
005
006import com.google.common.base.Objects;
007import com.google.common.base.Optional;
008import org.codehaus.jackson.annotate.JsonCreator;
009import org.codehaus.jackson.annotate.JsonProperty;
010
011import static com.google.common.base.Preconditions.checkArgument;
012import static com.google.common.base.Preconditions.checkNotNull;
013
014/**
015 * We send this message every time we encounter an error during crawling.
016 */
017public class CrawlErrorMessage implements DatasetBasedMessage {
018
019  private final UUID datasetUuid;
020
021  private final int attempt;
022
023  private final int retry;
024
025  private final int offset;
026
027  private final String status;
028
029  private final ErrorType errorType;
030
031  private final Optional<Throwable> throwable;
032
033  private final long duration;
034
035  @JsonCreator
036  public CrawlErrorMessage(
037    @JsonProperty("datasetUuid") UUID datasetUuid,
038    @JsonProperty("attempt") int attempt,
039    @JsonProperty("retry") int retry,
040    @JsonProperty("offset") int offset,
041    @JsonProperty("duration") long duration,
042    @JsonProperty("status") String status,
043    @JsonProperty("errorType") ErrorType errorType,
044    @Nullable @JsonProperty("throwable") Throwable throwable
045  ) {
046    this.duration = duration;
047    this.datasetUuid = checkNotNull(datasetUuid);
048    checkArgument(attempt > 0, "attempt has to be greater than 0");
049    this.attempt = attempt;
050    checkArgument(retry > 0, "retry has to be greater than 0");
051    this.retry = retry;
052    checkArgument(offset >= 0, "offset has to be greater than or equal to 0");
053    this.offset = offset;
054    this.status = checkNotNull(status);
055    this.errorType = checkNotNull(errorType);
056    this.throwable = Optional.fromNullable(throwable);
057  }
058
059  public int getAttempt() {
060    return attempt;
061  }
062
063  @Override
064  public UUID getDatasetUuid() {
065    return datasetUuid;
066  }
067
068  public long getDuration() {
069    return duration;
070  }
071
072  public ErrorType getErrorType() {
073    return errorType;
074  }
075
076  public int getOffset() {
077    return offset;
078  }
079
080  public int getRetry() {
081    return retry;
082  }
083
084  public String getStatus() {
085    return status;
086  }
087
088  @Override
089  public String getRoutingKey() {
090    return "crawl.error" + errorType.toString().toLowerCase();
091  }
092
093  public Optional<Throwable> getThrowable() {
094    return throwable;
095  }
096
097  @Override
098  public boolean equals(Object obj) {
099    if (obj == null) {
100      return false;
101    }
102    if (getClass() != obj.getClass()) {
103      return false;
104    }
105    final CrawlErrorMessage other = (CrawlErrorMessage) obj;
106    return Objects.equal(this.datasetUuid, other.datasetUuid)
107           && Objects.equal(this.attempt, other.attempt)
108           && Objects.equal(this.retry, other.retry)
109           && Objects.equal(this.offset, other.offset)
110           && Objects.equal(this.status, other.status)
111           && Objects.equal(this.errorType, other.errorType)
112           && Objects.equal(this.duration, other.duration);
113  }
114
115  @Override
116  public int hashCode() {
117    return Objects.hashCode(datasetUuid, attempt, retry, offset, status, errorType, duration);
118  }
119
120  @Override
121  public String toString() {
122    return Objects.toStringHelper(this)
123      .add("datasetUuid", datasetUuid)
124      .add("attempt", attempt)
125      .add("retry", retry)
126      .add("offset", offset)
127      .add("status", status)
128      .add("errorType", errorType)
129      .add("throwable", throwable)
130      .add("duration", duration)
131      .toString();
132  }
133
134  public enum ErrorType {
135
136    /**
137     * A protocol error means some kind of problem with the response we got. This can be invalid XML or things that
138     * just don't make sense.
139     */
140    PROTOCOL,
141
142    /**
143     * A transport error is any kind of error that's related to the transport layer like a refused connection or a 404
144     * error etc.
145     */
146    TRANSPORT,
147
148    /**
149     * A different error which we know what it is but doesn't fit into any of the other two categories.
150     */
151    OTHER,
152
153    /**
154     * An error that we didn't expect and have no idea what it is.
155     */
156    UNKNOWN
157
158  }
159
160}
