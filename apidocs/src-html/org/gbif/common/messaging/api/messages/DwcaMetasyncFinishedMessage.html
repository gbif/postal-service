001package org.gbif.common.messaging.api.messages;
002
003import org.gbif.api.model.crawler.DwcaValidationReport;
004import org.gbif.api.vocabulary.DatasetType;
005
006import java.net.URI;
007import java.util.Map;
008import java.util.UUID;
009
010import com.google.common.base.Objects;
011import org.codehaus.jackson.annotate.JsonCreator;
012import org.codehaus.jackson.annotate.JsonProperty;
013
014import static com.google.common.base.Preconditions.checkArgument;
015import static com.google.common.base.Preconditions.checkNotNull;
016
017/**
018 * We send this every time the dataset metadata found in a darwin core archive has been
019 * deposited in the metadata repository. This includes updating the registered dataset metadata and
020 * synchronizing potentially all found dataset constituents, e.g. GSDs in CoL.
021 */
022public class DwcaMetasyncFinishedMessage implements DatasetBasedMessage {
023
024  public static final String ROUTING_KEY = "crawl.dwca.metasync.finished";
025
026  private final UUID datasetUuid;
027  private final DatasetType datasetType;
028  private final URI source;
029  private final int attempt;
030  private final Map<String, UUID> constituents;
031  private final DwcaValidationReport validationReport;
032
033  @JsonCreator
034  public DwcaMetasyncFinishedMessage(
035    @JsonProperty("datasetUuid") UUID datasetUuid,
036    @JsonProperty("datasetType") DatasetType datasetType,
037    @JsonProperty("source") URI source,
038    @JsonProperty("attempt") int attempt,
039    @JsonProperty("constituents") Map<String, UUID> constituents,
040    @JsonProperty("validationReport") DwcaValidationReport validationReport
041  ) {
042    this.datasetUuid = checkNotNull(datasetUuid, "datasetUuid can't be null");
043    this.datasetType = checkNotNull(datasetType, "datasetType can't be null");
044    this.source = checkNotNull(source, "source can't be null");
045    checkArgument(attempt > 0, "attempt has to be greater than 0");
046    this.attempt = attempt;
047    this.constituents = checkNotNull(constituents, "constituents can't be null");
048    this.validationReport = checkNotNull(validationReport, "validationReport can't be null");
049  }
050
051  /**
052   * @return dataset uuid
053   */
054  @Override
055  public UUID getDatasetUuid() {
056    return datasetUuid;
057  }
058
059  /**
060   * @return the dataset type as given in the registry
061   */
062  public DatasetType getDatasetType() {
063    return datasetType;
064  }
065
066  /**
067   * @return source the archive has been downloaded from
068   */
069  public URI getSource() {
070    return source;
071  }
072
073  public int getAttempt() {
074    return attempt;
075  }
076
077  /**
078   * Defines a mapping of the dwc:datasetID used in the archive data and our UUID datasetUuid
079   * for all constituent (aka sub-) datasets found in this archive. Its a convenience map needed for indexing but
080   * could also be build by querying the registry constituent datasets and their tags.
081   *
082   * @return map of dwc:datasetID to datasetUuid
083   */
084  public Map<String, UUID> getConstituents() {
085    return constituents;
086  }
087
088  /**
089   * @return the validationReport for this archive
090   */
091  public DwcaValidationReport getValidationReport() {
092    return validationReport;
093  }
094
095  @Override
096  public String getRoutingKey() {
097    return ROUTING_KEY;
098  }
099
100  @Override
101  public int hashCode() {
102    return Objects.hashCode(datasetUuid, datasetType, source, attempt, constituents, validationReport);
103  }
104
105  @Override
106  public boolean equals(Object obj) {
107    if (this == obj) {
108      return true;
109    }
110    if (obj == null || getClass() != obj.getClass()) {
111      return false;
112    }
113    final DwcaMetasyncFinishedMessage other = (DwcaMetasyncFinishedMessage) obj;
114    return Objects.equal(this.datasetUuid, other.datasetUuid)
115           && Objects.equal(this.datasetType, other.datasetType)
116           && Objects.equal(this.source, other.source)
117           && Objects.equal(this.attempt, other.attempt)
118           && Objects.equal(this.constituents, other.constituents)
119           && Objects.equal(this.validationReport, other.validationReport);
120  }
121
122  @Override
123  public String toString() {
124    return Objects.toStringHelper(this)
125      .add("datasetUuid", datasetUuid)
126      .add("datasetType", datasetType)
127      .add("source", source)
128      .add("attempt", attempt)
129      .add("constituents", constituents)
130      .add("validationReport", validationReport)
131      .toString();
132  }
133}
