001package org.gbif.common.messaging.api.messages;
002
003import java.util.UUID;
004
005import com.google.common.base.Objects;
006import org.codehaus.jackson.annotate.JsonCreator;
007import org.codehaus.jackson.annotate.JsonProperty;
008
009import static com.google.common.base.Preconditions.checkNotNull;
010
011/**
012 * This message instructs the occurrence deletion service to delete all occurrence records for the given dataset.
013 */
014public class DeleteDatasetOccurrencesMessage implements DatasetBasedMessage {
015
016  private final UUID datasetUuid;
017  private final OccurrenceDeletionReason deletionReason;
018
019  @JsonCreator
020  public DeleteDatasetOccurrencesMessage(@JsonProperty("datasetUuid") UUID datasetUuid,
021    @JsonProperty("deletionReason") OccurrenceDeletionReason deletionReason) {
022    this.datasetUuid = checkNotNull(datasetUuid, "datasetUuid can't be null");
023    this.deletionReason = checkNotNull(deletionReason, "deletionReason can't be null");
024  }
025
026  @Override
027  public String getRoutingKey() {
028    return "occurrence.delete.dataset";
029  }
030
031  @Override
032  public UUID getDatasetUuid() {
033    return datasetUuid;
034  }
035
036  public OccurrenceDeletionReason getDeletionReason() {
037    return deletionReason;
038  }
039
040  @Override
041  public int hashCode() {
042    return Objects.hashCode(datasetUuid);
043  }
044
045  @Override
046  public boolean equals(Object obj) {
047    if (obj == null) {
048      return false;
049    }
050    if (getClass() != obj.getClass()) {
051      return false;
052    }
053    final DeleteDatasetOccurrencesMessage other = (DeleteDatasetOccurrencesMessage) obj;
054    return Objects.equal(this.datasetUuid, other.datasetUuid);
055  }
056}
