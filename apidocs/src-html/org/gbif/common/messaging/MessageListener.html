001package org.gbif.common.messaging;
002
003import org.gbif.common.messaging.api.Message;
004import org.gbif.common.messaging.api.MessageCallback;
005import org.gbif.common.messaging.api.MessageRegistry;
006
007import java.io.IOException;
008import java.util.concurrent.Executors;
009
010import com.fasterxml.jackson.datatype.guava.GuavaModule;
011import com.google.common.base.Optional;
012import com.rabbitmq.client.Channel;
013import com.rabbitmq.client.Connection;
014import com.rabbitmq.client.ConnectionFactory;
015import org.codehaus.jackson.map.ObjectMapper;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019import static com.google.common.base.Preconditions.checkArgument;
020import static com.google.common.base.Preconditions.checkNotNull;
021
022public class MessageListener {
023
024  private static final Logger LOG = LoggerFactory.getLogger(MessageListener.class);
025  private int prefetchCount = 50;
026  private final ConnectionFactory connectionFactory;
027  private final MessageRegistry registry;
028  private final ObjectMapper mapper;
029
030  /**
031   * Convenience constructor that uses a default {@link ObjectMapper} and the {@link DefaultMessageRegistry}.
032   */
033  public MessageListener(ConnectionParameters connectionParameters) throws IOException {
034    this(connectionParameters, new DefaultMessageRegistry(), new ObjectMapper());
035  }
036
037  /**
038   * Convenience constructor that uses a default {@link ObjectMapper} and the {@link DefaultMessageRegistry} but allows
039   * setting of the prefetch count (the RabbitMQ QOS).
040   * <br/>
041   * The prefetch count is how many records a channel will batch consume at a time.  Thus if there are long running
042   * consumers, and slow message throughput, one might opt to run a low prefetch count (like 1), in order to allow
043   * other channels to consume messages.  This blog is worth reading before opting to change the default prefetch count
044   * <a href="http://www.rabbitmq.com/blog/2012/05/11/some-queuing-theory-throughput-latency-and-bandwidth/">
045   * http://www.rabbitmq.com/blog/2012/05/11/some-queuing-theory-throughput-latency-and-bandwidth/</a>
046   */
047  public MessageListener(ConnectionParameters connectionParameters, int prefetchCount) throws IOException {
048    this(connectionParameters, new DefaultMessageRegistry(), new ObjectMapper());
049    checkArgument(prefetchCount >= 1, "prefetchCount needs to be greater than or equal to 1");
050    this.prefetchCount = prefetchCount;
051  }
052
053  /**
054   * Builds a new MessagingService with the provided components.
055   */
056  public MessageListener(ConnectionParameters connectionParameters, MessageRegistry registry, ObjectMapper mapper)
057    throws IOException {
058    checkNotNull(connectionParameters, "connectionParameters can't be null");
059    this.mapper = checkNotNull(mapper, "mapper can't be null");
060    this.registry = checkNotNull(registry, "registry can't be null");
061
062    connectionFactory = connectionParameters.getConnectionFactory();
063    // This ensures that the connection is valid, otherwise it'd throw an exception now
064    Connection connection = connectionFactory.newConnection();
065    connection.close(); // we don't store this or reuse it
066    mapper.registerModule(new GuavaModule());
067  }
068
069  /**
070   * Can be used to listen for a certain type of message. This will use the default generic routing key to bind to the
071   * default exchange as provided from the message registry.
072   *
073   * @param queue    to bind to
074   * @param callback to execute
075   * @param <T>      of the message
076   *
077   * @throws IOException if there was any communication exception
078   * @see #listen(String, String, String, int, MessageCallback)
079   */
080  public <T extends Message> void listen(String queue, int numberOfThreads, MessageCallback<T> callback)
081    throws IOException {
082    checkNotNull(callback, "callback can't be null");
083
084    Optional<String> routingKey = registry.getGenericRoutingKey(callback.getMessageClass());
085    checkArgument(routingKey.isPresent(), "The message needs to be registered");
086
087    listen(queue, routingKey.get(), numberOfThreads, callback);
088  }
089
090  /**
091   * Can be used to listen for a certain type of message. This will use the provided routing key to bind to the
092   * default exchange as provided from the message registry.
093   *
094   * @param queue      to bind to
095   * @param routingKey to use to bind the queue to the default exchange for this type of message
096   * @param callback   to execute
097   * @param <T>        of the message
098   *
099   * @throws IOException if there was any communication exception
100   * @see #listen(String, String, String, int, MessageCallback)
101   */
102  public <T extends Message> void listen(String queue, String routingKey, int numberOfThreads,
103    MessageCallback<T> callback) throws IOException {
104    checkNotNull(callback, "callback can't be null");
105
106    Optional<String> exchange = registry.getExchange(callback.getMessageClass());
107    checkArgument(exchange.isPresent(), "The message needs to be registered");
108
109    listen(queue, routingKey, exchange.get(), numberOfThreads, callback);
110  }
111
112  /**
113   * Can be used to listen for a certain type of message. Prior to registering the callback it will declare the
114   * exchange and queue and bind the queue to the exchange using the provided routing key.
115   * The callback will be automatically invoked for each new message. Each message will be acknowledged as soon as the
116   * message is delivered (not processed). Uses the default prefetch count (messages to prefetch from the queue).
117   *
118   * @param queue           to bind to
119   * @param routingKey      to use to bind the queue to the exchange
120   * @param exchange        to bind to
121   * @param numberOfThreads the number of threads to use to process messages for this callback in parallel
122   * @param callback        to execute
123   * @param <T>             of the message
124   *
125   * @throws IOException if there was any communication exception
126   */
127  public <T> void listen(String queue, String routingKey, String exchange, int numberOfThreads,
128    MessageCallback<T> callback) throws IOException {
129    checkNotNull(queue, "queue can't be null");
130    checkNotNull(routingKey, "routingKey can't be null");
131    checkNotNull(callback, "callback can't be empty");
132    checkArgument(numberOfThreads >= 1, "numberOfThreads needs to be greater than or equal to 1");
133
134    Connection connection = connectionFactory.newConnection(Executors.newFixedThreadPool(numberOfThreads));
135    Channel channel = connection.createChannel();
136    channel.exchangeDeclare(exchange, "topic", true);
137    channel.queueDeclare(queue, true, false, false, null);
138    channel.queueBind(queue, exchange, routingKey);
139    channel.close();
140
141    LOG
142      .debug("Starting to listen on exchange [{}], queue [{}] and routing key [{}] for messages of type [{}]", exchange,
143        queue, routingKey, callback.getMessageClass().getSimpleName());
144
145    for (int i = 0; i < numberOfThreads; i++) {
146      channel = connection.createChannel();
147      channel.basicQos(prefetchCount);
148      channel.basicConsume(queue, false, // autoAck disabled -> we have to manually acknowledge messages
149        new MessageConsumer<T>(callback.getMessageClass(), channel, mapper, callback));
150    }
151  }
152
153  /**
154   * Does nothing, but adding for potential future use.
155   */
156  public void close() {
157
158  }
159}
