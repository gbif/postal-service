001package org.gbif.common.messaging;
002
003import org.gbif.common.messaging.api.MessageCallback;
004
005import java.lang.reflect.ParameterizedType;
006import java.lang.reflect.Type;
007
008/**
009 * Implements the {@link MessageCallback} interface and provides a convenience implementation of the {@link
010 * #getMessageClass()} method.
011 */
012public abstract class AbstractMessageCallback<T> implements MessageCallback<T> {
013
014  // Taken from: http://stackoverflow.com/questions/182636/how-to-determine-the-class-of-a-generic-type
015  @Override
016  public Class<T> getMessageClass() {
017    Class<?> superClass = getClass(); // initial value
018    Type superType;
019    do {
020      superType = superClass.getGenericSuperclass();
021      superClass = extractClassFromType(superType);
022    } while (!superClass.equals(AbstractMessageCallback.class));
023
024    Type actualArg = ((ParameterizedType) superType).getActualTypeArguments()[0];
025    return (Class<T>) extractClassFromType(actualArg);
026  }
027
028  private Class<?> extractClassFromType(Type t) throws ClassCastException {
029    if (t instanceof Class<?>) {
030      return (Class<?>) t;
031    }
032    return (Class<?>) ((ParameterizedType) t).getRawType();
033  }
034
035}
