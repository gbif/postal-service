37     * new channels are created. This is unbounded in size and dependent on consumer load (e.g. concurrent threads).

39    private final ConcurrentLinkedQueue<Channel> channelPool = Queues.newConcurrentLinkedQueue();

91    @Override
92    public void send(Object message, String exchange, String routingKey) throws IOException {
93      checkNotNull(message, "message can't be null");
94      checkNotNull(exchange, "exchange can't be null");
95      checkNotNull(routingKey, "routingKey can't be null");
96  
97      byte[] data = mapper.writeValueAsBytes(message);
98      LOG.debug("Sending message of type [{}] to exchange [{}] using routing key [{}]",
99                message.getClass().getSimpleName(),
100               exchange,
101               routingKey);
102 
103     for (int attempt = 1; attempt <= NUMBER_OF_RETRIES; attempt++) {
104       Channel channel = provideChannel();
105 
106       try {
107         channel.basicPublish(exchange, routingKey, MessageProperties.TEXT_PLAIN, data);
108         // We're not releasing this in a finally block because we assume the channel is "bad" if an exception occurred
109         releaseChannel(channel);
110         return;
111       } catch (IOException e) {
112         if (attempt >= NUMBER_OF_RETRIES) {
113           LOG.warn("Tried sending message but failed {} times, aborting", attempt);
114           throw e;
115         }
116         LOG.debug("Failed sending message, retrying");
117       }
118     }
119 
120   }

122   /**
123    * Declares all exchanges used by the registry as <em>topic</em> exchanges.
124    *
125    * @param registry   to get the registered messages and exchanges from
126    * @param connection used to declare the exchanges
127    *
128    * @throws IOException if there was a problem declaring an exchange. This might leave the exchanges partially
129    *                     declared
130    */
131   private void declareAllExchanges(MessageRegistry registry, Connection connection) throws IOException {
132     Channel channel = null;
133     try {
134       channel = connection.createChannel();
135       for (Class<? extends Message> message : registry.getRegisteredMessages()) {
136         channel.exchangeDeclare(registry.getExchange(message).get(), "topic", true);
137       }
138     } finally {
139       if (channel != null) {
140         channel.close();
141       }
142     }
143   }

145   /**
146    * Callers must return the channel using {@link #releaseChannel(Channel channel) releaseChannel}.
147    *
148    * @return a channel to reuse, or a new one if none are available
149    */
150   private Channel provideChannel() throws IOException {
151     Channel channel = channelPool.poll();
152     if (channel == null || !channel.isOpen()) { // none for reuse
153       LOG.debug("No pooled channels available, creating a new one");
154       channel = connection.createChannel();
155     }
156     return channel;
157   }
158 
159   /**
160    * Returns the channels to the pool for reuse.
161    *
162    * @param channel To free up for another use
163    */
164   private void releaseChannel(Channel channel) {
165     // only reuse open channels
166     if (channel.isOpen()) {
167       channelPool.add(channel);
168       LOG.debug("Channel returned to the pool. Available channels for reuse: {}", channelPool.size());
169     } else {
170       LOG.debug("Discarding channel since it is closed. Available channels for reuse: {}", channelPool.size());
171     }
172   }
