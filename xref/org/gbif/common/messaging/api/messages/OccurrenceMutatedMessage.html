16  /**
17   * The message sent whenever an "interpreted" occurrence has changed (either NEW, UPDATED, or DELETED). An interpreted
18   * occurrence is the result of interpreting the values of a verbatim occurrence.
19   */

27    private final Occurrence newOccurrence;

34    // the latest crawl attempt for this occurrence's dataset

37    @JsonCreator
38    public OccurrenceMutatedMessage(@JsonProperty("datasetUuid") UUID datasetUuid,
39      @JsonProperty("status") OccurrencePersistenceStatus status,
40      @Nullable @JsonProperty("crawlAttempt") Integer crawlAttempt,
41      @Nullable @JsonProperty("oldOccurrence") Occurrence oldOccurrence,
42      @Nullable @JsonProperty("newOccurrence") Occurrence newOccurrence,
43      @Nullable @JsonProperty("deletionReason") OccurrenceDeletionReason deletionReason,
44      @Nullable @JsonProperty("crawlAttemptLastSeen") Integer crawlAttemptLastSeen,
45      @Nullable @JsonProperty("latestCrawlAttemptForDataset") Integer latestCrawlAttemptForDataset) {
46      this.datasetUuid = checkNotNull(datasetUuid, "datasetUuid can't be null");
47      this.status = checkNotNull(status, "status can't be null");
48      switch (status) {
49        case NEW:
50          checkNotNull(newOccurrence, "newOccurrence can't be null");
51          checkArgument(crawlAttempt != null && crawlAttempt > 0, "attempt must be greater than 0");
52          break;
53        case UPDATED:
54          checkNotNull(newOccurrence, "newOccurrence can't be null for updates");
55          checkNotNull(oldOccurrence, "oldOccurrence can't be null for updates");
56          checkArgument(crawlAttempt != null && crawlAttempt > 0, "attempt must be greater than 0");
57          break;
58        case DELETED:
59          checkNotNull(oldOccurrence, "oldOccurrence can't be null for deletes");
60          checkNotNull(deletionReason, "deletionReason can't be null for deletes");
61          if (deletionReason == OccurrenceDeletionReason.NOT_SEEN_IN_LAST_CRAWL) {
62            checkArgument(crawlAttemptLastSeen != null && crawlAttemptLastSeen > 0,
63              "crawlAttemptLastSeen must be greater than 0");
64            checkArgument(latestCrawlAttemptForDataset != null && latestCrawlAttemptForDataset > 0,
65              "latestCrawlAttemptForDataset must be greater than 0");
66          }
67          break;
68        case UNCHANGED: // should never be seen
69          break;
70      }
71      this.crawlAttempt = crawlAttempt;
72      this.newOccurrence = newOccurrence;
73      this.oldOccurrence = oldOccurrence;
74      this.deletionReason = deletionReason;
75      this.crawlAttemptLastSeen = crawlAttemptLastSeen;
76      this.latestCrawlAttemptForDataset = latestCrawlAttemptForDataset;
77    }
