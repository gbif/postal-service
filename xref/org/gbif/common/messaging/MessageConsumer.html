21  /**
22   * This is the consumer used by {@link MessageListener} to handle AMQP deliveries. It deserializes the content and
23   * forwards it to a callback. All messages are automatically either being rejected or acked.
24   * <p/>
25   * This class is thread-safe but stateful.
26   * 
27   * @param <T> type of message to handle, we'll try to deserialize the received content in an instance of this type
28   */
29  @ThreadSafe
30  class MessageConsumer<T> extends DefaultConsumer {
31  
32    private static final Logger LOG = LoggerFactory.getLogger(MessageConsumer.class);
33    private final MessageCallback<T> callback;
34    private final Class<T> clazz;
35    private final ObjectMapper mapper;
36  
37    MessageConsumer(Class<T> clazz, Channel channel, ObjectMapper mapper, MessageCallback<T> callback) {
38      super(channel);
39      checkNotNull(channel, "channel can't be null");
40  
41      this.clazz = checkNotNull(clazz, "clazz can't be null");
42      this.mapper = checkNotNull(mapper, "mapper can't be null");
43      this.callback = checkNotNull(callback, "callback can't be null");
44    }
45  
46    /**
47     * Delivery is being handled by deserializing the message body and creating a Runnable that's being put on the
48     * ExecutorService.
49     * <p/>
50     * If deserializing fails for any reason we reject the message and don't requeue it, apart from that we ack the
51     * message no matter the outcome of the callback.
52     * <p/>
53     * If submitting a job to the ExecutorService fails we send a rejection back to RabbitMQ.
54     */
55    @Override
56    public void handleDelivery(
57      String consumerTag, Envelope envelope, AMQP.BasicProperties properties, byte[] body
58      ) throws IOException {
59      LOG.debug("Handling delivery: [{}]", envelope.getDeliveryTag());
60  
61      T object = deserializeBody(envelope, body);
62      if (object == null) {
63        getChannel().basicReject(envelope.getDeliveryTag(), false);
64      } else {
65        handleCallback(envelope, object); // which will ACK or NACK
66      }
67    }
68  
69    /**
70     * This tries to deserialize the body of the message, returning the Message or null.
71     */
72    private T deserializeBody(Envelope envelope, byte[] body) throws IOException {
73      T object = null;
74      try {
75        object = mapper.readValue(body, clazz);
76      } catch (JsonMappingException e) {
77        LOG.warn("Could not map message. Supposed to be of type [{}]. Routing key [{}], exchange [{}]",
78          clazz.getSimpleName(),
79          envelope.getRoutingKey(),
80          envelope.getExchange(),
81          e);
82      } catch (JsonParseException e) {
83        LOG.warn("Could not parse body of message. Supposed to be of type [{}]. Routing key [{}], exchange [{}]",
84          clazz.getSimpleName(),
85          envelope.getRoutingKey(),
86          envelope.getExchange(),
87          e);
88      } catch (IOException e) {
89        LOG.warn("Unable to read message over network. Supposed to be of type [{}]. Routing key [{}], exchange [{}]",
90          clazz.getSimpleName(),
91          envelope.getRoutingKey(),
92          envelope.getExchange(),
93          e);
94      }
95      return object; // will be null on any error
96    }
97  
98    private void handleCallback(Envelope envelope, T object) {
99      // Handle the message and send a Nack if the Callback throws an Exception
100     try {
101       callback.handleMessage(object);
102 
103     } catch (Exception e) {
104       LOG.warn("Error handling message, will be acknowledged anyway and not retried", e);
105       try {
106         getChannel().basicNack(envelope.getDeliveryTag(), false, false);
107       } catch (IOException e1) {
108         LOG.warn("Failure acknowledging message [{}] of type [{}]",
109           envelope.getDeliveryTag(),
110           object.getClass().getSimpleName(),
111           e1);
112       }
113       return;
114     }
115 
116     // Handling itself was successful now try to ack the message
117     try {
118       getChannel().basicAck(envelope.getDeliveryTag(), false);
119     } catch (IOException e) {
120       LOG.warn("Failure acknowledging message [{}] of type [{}]",
121         envelope.getDeliveryTag(),
122         object.getClass().getSimpleName(),
123         e);
124     }
125   }
