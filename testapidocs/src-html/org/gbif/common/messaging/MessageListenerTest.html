001package org.gbif.common.messaging;
002
003import org.gbif.common.messaging.api.MessageRegistry;
004
005import java.io.IOException;
006import java.util.concurrent.ExecutorService;
007
008import com.rabbitmq.client.Channel;
009import com.rabbitmq.client.Connection;
010import com.rabbitmq.client.Consumer;
011import org.codehaus.jackson.map.ObjectMapper;
012import org.junit.Before;
013import org.junit.Test;
014import org.junit.runner.RunWith;
015import org.mockito.Mock;
016import org.mockito.runners.MockitoJUnitRunner;
017
018import static org.mockito.Matchers.any;
019import static org.mockito.Matchers.eq;
020import static org.mockito.Matchers.isA;
021import static org.mockito.Mockito.verify;
022import static org.mockito.Mockito.when;
023
024@RunWith(MockitoJUnitRunner.class)
025public class MessageListenerTest {
026
027  @Mock
028  Connection connection;
029
030  @Mock
031  Channel channel;
032
033  private MessageListener listener;
034
035  private static final String DEFAULT_EXCHANGE = "foo";
036  private static final String DEFAULT_ROUTINGKEY = "bar";
037  private static final String QUEUE = "queue";
038
039  @Before
040  public void setup() throws IOException {
041    MessageRegistry registry = new DefaultMessageRegistry();
042    registry.register(TestMessage.class, DEFAULT_EXCHANGE, DEFAULT_ROUTINGKEY);
043    when(connection.createChannel()).thenReturn(channel);
044    when(connection.isOpen()).thenReturn(true);
045
046    ConnectionParameters params = new MockConnectionParameters();
047    when(params.getConnectionFactory().newConnection(any(ExecutorService.class))).thenReturn(connection);
048    when(params.getConnectionFactory().newConnection()).thenReturn(connection);
049    listener = new MessageListener(params, registry, new ObjectMapper());
050  }
051
052  @Test
053  public void testListen1() throws IOException {
054    listener.listen(QUEUE, 1, new TestMessageCallback());
055
056    verify(channel).exchangeDeclare(DEFAULT_EXCHANGE, "topic", true);
057    verify(channel).queueDeclare(QUEUE, true, false, false, null);
058    verify(channel).queueBind(QUEUE, DEFAULT_EXCHANGE, DEFAULT_ROUTINGKEY);
059
060    verify(channel).basicConsume(eq(QUEUE), eq(false), isA(Consumer.class));
061  }
062
063  @Test
064  public void testListen2() throws IOException {
065    String queue = "foobar";
066    listener.listen(queue, "bar", 1, new TestMessageCallback());
067
068    verify(channel).exchangeDeclare(DEFAULT_EXCHANGE, "topic", true);
069    verify(channel).queueDeclare(queue, true, false, false, null);
070    verify(channel).queueBind(queue, DEFAULT_EXCHANGE, DEFAULT_ROUTINGKEY);
071
072    verify(channel).basicConsume(eq(queue), eq(false), isA(Consumer.class));
073  }
074
075  @Test
076  public void testListen3() throws IOException {
077    String queue = "foobar";
078    String exchange = "barxchange";
079
080    listener.listen(queue, "bar", exchange, 1, new TestMessageCallback());
081
082    verify(channel).exchangeDeclare(exchange, "topic", true);
083    verify(channel).queueDeclare(queue, true, false, false, null);
084    verify(channel).queueBind(queue, exchange, DEFAULT_ROUTINGKEY);
085
086    verify(channel).basicConsume(eq(queue), eq(false), isA(Consumer.class));
087  }
088
089  private static class TestMessageCallback extends AbstractMessageCallback<TestMessage> {
090
091    @Override
092    public void handleMessage(TestMessage message) {
093
094    }
095
096  }
097
098}
