001package org.gbif.common.messaging;
002
003import org.gbif.common.messaging.api.MessagePublisher;
004
005import java.io.IOException;
006
007import org.junit.Test;
008import org.junit.runner.RunWith;
009import org.mockito.Mock;
010import org.mockito.runners.MockitoJUnitRunner;
011
012import static org.mockito.Mockito.verify;
013
014@RunWith(MockitoJUnitRunner.class)
015public class ForwardingMessagePublisherTest {
016
017  private final TestMessage message = new TestMessage();
018
019  @Mock
020  private MessagePublisher mockPublisher;
021
022  @Test
023  public void testForwarding1() throws IOException {
024    MessagePublisher publisher = new ForwardingMessagePublisher(mockPublisher);
025    publisher.send(message);
026    verify(mockPublisher).send(message);
027  }
028
029  @Test
030  public void testForwarding2() throws IOException {
031    MessagePublisher publisher = new ForwardingMessagePublisher(mockPublisher);
032    publisher.send(message, "foobar");
033    verify(mockPublisher).send(message, "foobar");
034  }
035
036  @Test
037  public void testForwarding3() throws IOException {
038    MessagePublisher publisher = new ForwardingMessagePublisher(mockPublisher);
039    publisher.send(message, "foobar", "routing");
040    verify(mockPublisher).send(message, "foobar", "routing");
041  }
042
043}
