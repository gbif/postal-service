001package org.gbif.common.messaging.api.messages;
002
003import org.gbif.api.model.occurrence.Occurrence;
004import org.gbif.api.vocabulary.OccurrencePersistenceStatus;
005import org.gbif.api.vocabulary.EndpointType;
006import org.gbif.common.messaging.api.Util;
007
008import java.io.IOException;
009import java.util.UUID;
010
011import org.junit.Test;
012
013public class OccurrenceMutatedMessageTest {
014
015  @Test
016  public void testDefaultMessageRegistry() {
017    Util.testDefaultMessageRegistry(OccurrenceMutatedMessage.class);
018  }
019
020  @Test
021  public void testSerDe() throws IOException {
022    OccurrenceMutatedMessage message =
023      new OccurrenceMutatedMessage(UUID.randomUUID(), OccurrencePersistenceStatus.UPDATED, 1,
024      build(EndpointType.BIOCASE, 1, UUID.randomUUID()),
025      new Occurrence(), null, null, null);
026    Util.testSerDe(message, OccurrenceMutatedMessage.class);
027  }
028
029  @Test
030  public void testBuilders() throws IOException {
031    OccurrenceMutatedMessage msg = OccurrenceMutatedMessage.buildNewMessage(UUID.randomUUID(),
032      build(EndpointType.BIOCASE, 1, UUID.randomUUID()), 1);
033    Util.testSerDe(msg, OccurrenceMutatedMessage.class);
034
035    msg = OccurrenceMutatedMessage.buildUpdateMessage(UUID.randomUUID(),
036                                                      build(EndpointType.BIOCASE, 1, UUID.randomUUID()),
037                                                      build(EndpointType.BIOCASE, 1, UUID.randomUUID()), 1);
038    Util.testSerDe(msg, OccurrenceMutatedMessage.class);
039
040    msg = OccurrenceMutatedMessage.buildDeleteMessage(UUID.randomUUID(),
041      build(EndpointType.BIOCASE, 1, UUID.randomUUID()),
042      OccurrenceDeletionReason.OCCURRENCE_MANUAL, null, null);
043    Util.testSerDe(msg, OccurrenceMutatedMessage.class);
044  }
045
046  private Occurrence build(EndpointType prot, int key, UUID dkey) {
047    Occurrence o = new Occurrence();
048    o.setKey(key);
049    o.setProtocol(prot);
050    o.setDatasetKey(dkey);
051    return o;
052  }
053}
